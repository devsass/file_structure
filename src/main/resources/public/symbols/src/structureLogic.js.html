 18 function createFolder (divID) {
 19     var buttonName = 'button' + bCount;
 20     bCount += 1;
 21     var div = createDiv(buttonName + "div");
 22     var filesDiv = createDiv(buttonName + "div" + "files");
 23     var img = createImg(buttonName + "img", "14", "20", "index.png");
 24     var name = document.createTextNode(buttonName);
 25     var button = document.createElement("BUTTON");
 26     var dropB = createDropButton(buttonName + "drop", filesDiv.getAttribute("id"));
 27     var addB = createAddButton(buttonName + "add", filesDiv.getAttribute("id"));
 28     button.appendChild(name);
 29     document.body.appendChild(button);
 30     button.setAttribute("id", buttonName);
 31     button.setAttribute("class", "folderStyle");
 32     button.addEventListener("dblclick", openFile);
 33     div.appendChild(dropB);
 34     div.appendChild(img);
 35     div.appendChild(button);
 36     div.appendChild(addB);
 37     div.appendChild(filesDiv);
 38     document.getElementById(divID).appendChild(div);
 39 }

 59 function createDropButton(id, divId) {
 60     var dropB = document.createElement("BUTTON");
 61     dropB.innerHTML = ">";
 62     dropB.setAttribute("id", id);
 63     dropB.setAttribute("onclick", "toggleDisplay('"+ divId + "','" + id + "');");
 64     dropB.setAttribute("class", "dropdown");
 65     document.body.appendChild(dropB);
 66     return dropB;
 67 }

 69 function createAddButton(id, param) {
 70     var addB = document.createElement("BUTTON");
 71     addB.innerHTML = (createImg(id+"img", 10, 10, "add_file.png")).outerHTML;
 72     addB.setAttribute("id", id);
 73     addB.setAttribute("class", "addButton")
 74     addB.setAttribute("onclick", "createFile('" + param + "')");
 75     document.body.appendChild(addB);
 76     return addB;
 77 }

 79 function createImg (id, hi, wi, imgfile) {
 80     var img = document.createElement("img");
 81     img.setAttribute("id", id);
 82     img.setAttribute("height", hi);
 83     img.setAttribute("width", wi);
 84     img.setAttribute("src", imgfile);
 85     document.body.appendChild(img);
 86     return img;
 87 }

102 function toggleDisplay(id, id2) {
103     if (document.getElementById(id).style.display == "block") {
104         document.getElementById(id).style.display = "none";
105         document.getElementById(id2).innerHTML = ">";
106     } else {
107         document.getElementById(id).style.display = "block";
108         document.getElementById(id2).innerHTML = "^";
109     }
110 }
